Day 03 - Ansible, Terrafrom & Kubernetes Basics

  Ansible - Install Apache and Start Service
-name: Install and start Apache
 host: all
 become: all
 tasks:
   - name: Install apache
     yum:
       name: httpd
       state: present


    - name: Start apache
      Service:
         name: httpd
	 state: sarted
	 enabled: yes

# Kubernetes - Create a ReplicaSEt with 3 Replicas
apiVersions: apps/v1
kind: ReplicaSet
metadata:
    name: pathnex-replicaset
spec:
  replicas: 3
  selector:
     matchLabels:
        app: pathnex-app
  template:
     metadata:
        labels:
	  app: pathnex-app
      spec:
        containers:
	   - name: app
	     image: nginx
	     ports:
	        - containerPort: 80
