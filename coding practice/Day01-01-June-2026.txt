## Kubernetes Task - Create Nginx Pod

apiVersion: v1
kind: Pod
metadata:
   name: pathnex-pod
spec:
  containers:
    - name: web
      image: nginx: latest
      ports:
        - containerPort: 80

# Git Integration
## Jenkins Pipwline -Checkout Git

pipeline {
    agent any
    stages {
       stage('Checkout') {
          steps {
	    git branch: 'main', url: 'https://github.com/Pathnex/sample-java-app.git'
	  }
       }
       stage('List Files') {
          steps {
	     sh 'ls -la'
	  }
       }
    }
}



## GitLab CI - Checkout Git
satges:
   - checkout


git-checkout:
   stage: checkout
   script:
      - git clone https://github.com/Pathnex/sample-java-app.git
      - cd sample-java-app
      - ls -la
